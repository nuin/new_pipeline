import os
import subprocess
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)



[docs]def call_qualimap(sample_id, directory, bed_file, qualimap):
    """
    Function that performs the call to qualimap

    :param sample_id: ID of the patient/sample being analysed
    :param directory: Location of the BAM files
    :param bed_file: BED file with regions to be analysed
    :param qualimap: Qualimap jar/bat/etc location

    :type sample_id: string
    :type directory: string
    :type bed_file: string
    :type qualimap: string

    :return: returns success or exists

    :todo: return error
    :todo: fix argument
    """

    argument = directory + '/BAM/' + sample_id + '/BAM/' + sample_id
    argument2 = directory + '/BAM/' + sample_id + '/QC/'

    if os.path.isfile(argument2 + '/QC/qualimapReport.html'):
        logger.info('Qualimap result exists')
        return 'exists'
    else:
        logger.info('Starting Qualimap report generation')
        qualimap_string = '%s bamqc -bam %s.recal_reads.bam -gff %s -c -nw 400 -hm 3 -outdir %s/QC/' % (qualimap, argument, bed_file, argument2)
        logger.info(qualimap_string)
        proc = subprocess.Popen(qualimap_string, shell=True, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
        while True:
            output = proc.stdout.readline().strip()
            if output == b'':
                break
            else:
                logger.info(output.decode('utf-8'))
        proc.wait()
        logger.info('Qualimap run successful')
        return 'success'
