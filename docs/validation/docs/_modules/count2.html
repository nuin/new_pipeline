import os
import pysam
import logging
import graypy

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)
graypy_handler = graypy.GELFUDPHandler('192.168.1.169', 12201)
logger.addHandler(graypy_handler)


[docs]def extract_counts(datadir, full_BED, sample_id):
    """
    Function that reads the BAM file and extract the read count for each window

    :param datadir: Location of the BAM files
    :param full_BED: BED file to guide the counts
    :param sample_id: ID of the sdample

    :type datadir: string
    :type full_BED: string
    :type sample_id: string

    :return: no return yet
    :todo: change BED file location from hardcoded
    """

    bedfile = open(full_BED).read().splitlines()

    if not os.path.isfile(datadir + '/BAM/' + sample_id + '/' + sample_id + '.bam'):
        sample = datadir + '/BAM/' + sample_id + '/BAM/' + sample_id + '.bam'
    else:
        sample = datadir + '/BAM/' + sample_id + '/' + sample_id + '.bam'

    if not os.path.isfile(datadir + '/BAM/' + sample_id + '/' + sample_id + '.cnv'):
        try:
            logger.info('Creating samples CNV file ' + datadir)
            cnv_out = open(datadir + '/BAM/' + sample_id + '/' + sample_id + '.cnv', 'w')
            cnv_out.write('Location\t')
            cnv_out.write(sample_id + '\n')

            logger.info('Analysing BAM file ' + sample + ' ' + datadir)
            samfile = pysam.AlignmentFile(sample, 'rb')
            for location in bedfile:
                temp = location.split('\t')
                cnv_out.write(temp[3] + '\t' + str(samfile.count(reference=temp[0], start=int(temp[1]), end=int(temp[2]))) + '\n')
            cnv_out.close()

            logger.info('BAM file analysed, CNV file created ' + sample + ' ' + datadir)
        except Exception as e:
            logger.error(str(e))
            logger.error('BAM file not found ' + sample + ' ' + datadir)
    else:
        logger.info('CNV file already exists')
