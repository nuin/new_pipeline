[docs]def get_annotation(sample_id, datadir, annovar):
    """
    Function that runs Annovar to annotate the merged VCF file

    :param sample_id: ID of the sample being analysed
    :param datadir: run location
    :param annovar: location of the annovar executable (Perl)

    :type sample_id: string
    :type datadir: string
    :type annovar: string

    :todo improve return
    :todo improve output
    :todo parse and DB connection
    """

    code = get_code(sample_id)
    argument_vcf = datadir + "/BAM/" + sample_id + "/VCF_" + code + "/" + sample_id

    if os.path.isfile(argument_vcf + ".hg19_multianno.vcf"):
        logger.info('Annovar annotated VCF exists ' + sample_id)
        return "exists"

    logger.info('Starting annotation with Annovar ' + sample_id)
    annovar_string = "%s/table_annovar.pl -vcfinput %s_merged.vcf %s/humandb/ -buildver hg19 -out %s -remove -protocol " \
                     "refGene,cytoBand,genomicSuperDups,esp6500siv2_all,1000g2015aug_all,1000g2015aug_eur,exac03," \
                     "avsnp147,dbnsfp35c,clinvar_20190305,cosmic70,gnomad211_exome,intervar_20180118,dbscsnv11,snp138," \
                     "revel,dbnsfp35a -operation g,r,r,f,f,f,f,f,f,f,f,f,f,f,f,f,f -nastring ." % (annovar,
                                                                                                   argument_vcf,
                                                                                                   annovar, argument_vcf)

    logger.info('Command ' + annovar_string + ' ' + sample_id)
    proc = subprocess.Popen(annovar_string, shell=True, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
    while True:
        output = proc.stdout.readline().strip()
        if output == b'':
            break
        else:
            logger.info(output)
    proc.wait()
    logger.info('Annovar annotated VCF created')

    return "success"
