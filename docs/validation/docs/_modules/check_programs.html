[docs]def cmd_exists(cmd):
    """
    Function that runs the defined command and returns its result

    :param cmd: Command to be tested
    :type cmd: string

    :return: the result of a subprocess.call operation with the desired command
    """

    return subprocess.call('type ' + cmd, shell=True, stdout=subprocess.PIPE,
                           stderr=subprocess.PIPE) == 0




[docs]def get_version(cmd):
    """
    Function that tries to extract the version of external programs required for analysis

    :param cmd: Command to be checked
    :type cmd: string

    :return: returns the standard output and standard error of a subprocess.Popen call
    :todo: log the return
    """

    proc = subprocess.Popen(
        cmd, shell=True, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
    std_err = proc.stderr.read()
    std_out = proc.stdout.read()

    return std_out + std_err




[docs]def find_version(stream):
    """
    Function that tries to find the external program version from either stdout or sterr outputs

    :param stream: stdout and stderr combined in a single stream
    :type stream: string

    :return: returns the line containing the word version
    :rtype: string
    """

    for line in stream.splitlines():
        if line.lower().find('version') >= 0:
            return line




[docs]def versions():
    """
    Main function of the script where the required external apps/languages are listed and called
    No parameters

    :return: returns a dictionary with the program and its version
    """

    to_return = {}
    if cmd_exists('picard'):
        to_return['picard'] = find_version(get_version('picard MarkDuplicates')).replace('Version: ', '')

    if cmd_exists('freebayes'):
        to_return['freebayes'] = find_version(get_version('freebayes --version')).replace('version:  ', '')

    if cmd_exists('java -jar /usr/local/bin/GenomeAnalysisTK.jar'):
        to_return['GATK'] = get_version('java -jar /usr/local/bin/GenomeAnalysisTK.jar --version').strip()

    if cmd_exists('/opt/Platypus_0.8.1/Platypus.py'):
        to_return['platypus'] = True

    if cmd_exists('bwa'):
        to_return['bwa'] = find_version(get_version('bwa')).replace('Version: ', '')

    if cmd_exists('ruby /opt/Atlas-Indel2/Atlas-Indel2.rb'):
        to_return['atlas'] = True

    return to_return

    """
    Function that checks the locations from the required external software based on a YAML
    configuraiton file

    :param software_conf: output from a YAML file stream with the key/value of required software

    :return: returns a dictionary with program:"found" or "error"

    to_return = {}

        to_return['BWA'] = 'error'

    if cmd_exists(software_conf['Picard']):
        to_return['Picard'] = 'found'
    else:
        to_return['Picard'] = 'error'

    if cmd_exists(software_conf['GATK']):
        to_return['GATK'] = 'found'
    else:
        to_return['GATK'] = 'error'
